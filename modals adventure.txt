<script>
    function checkAnswers() {
        const answers = {
            q1: "must",
            q2: "must not",
            q3: "could",
            q4: "can",
            q5: "ought to",
            q6: "need to"
        };

        let score = 0;
        let total = 6;

        for (let i = 1; i <= total; i++) {
            let selected = document.getElementById('q' + i).value;
            if (selected === answers['q' + i]) {
                score++;
            }
        }

        const resultDiv = document.getElementById('result');
        if (score === total) {
            resultDiv.innerHTML = `<span style="color: var(--primary-neon)">ACCESS GRANTED! 🔑 Score: ${score}/${total}. You escaped!</span>`;
        } else {
            resultDiv.innerHTML = `<span style="color: var(--danger-neon)">ACCESS DENIED! ❌ Score: ${score}/${total}. Check your Modals and try again!</span>`;
        }
    }
</script>
